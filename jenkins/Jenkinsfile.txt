pipeline {
    agent any
    stages {
        stage('clean') {
            steps {
				def mvnHome = tool name: 'maven', type: 'maven'
				def mvnCMD = "${mvnHome}/bin/mvn"
                bat "${mvnCMD} -f ../security/security/pom.xml clean";
            }
        }
		stage('test') {
            steps {
                bat "mvn test";
            }
        }
		stage('package') {
            steps {
                bat "mvn package";
            }
        }
    }
}
