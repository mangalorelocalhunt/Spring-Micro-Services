pipeline {
    agent any
	
	 environment {
        mvnHome = tool name: 'maven', type: 'maven'
		mvnCMD = "${mvnHome}/bin/mvn"
    }

	
    stages {
        stage('clean') {
            steps {
                bat "${mvnCMD} -f ../security/security/pom.xml clean";
            }
        }
		stage('test') {
            steps {
                bat "mvn test";
            }
        }
		stage('package') {
            steps {
                bat "mvn package";
            }
        }
    }
}
